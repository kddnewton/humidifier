def initialize(opts = {})
  PROPERTIES.each_value do |property|
    instance_variable_set(:"@#{property}", opts[property])
  end

  @type = opts.fetch(:type, "String")
end

def to_cf
  { "Type" => type }.tap do |cf|
    PROPERTIES.each do |name, prop|
      value = public_send(prop)
      cf[name] = Serializer.dump(value) if value
    end
  end
end
