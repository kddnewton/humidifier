def pretty_print(q)
  q.group do
    q.text("(#{name}=structure")
    q.nest(2) do
      q.breakable
      q.seplist(subprops.values) { |subprop| q.pp(subprop) }
    end
    q.breakable("")
    q.text(")")
  end
end

def to_cf(struct)
  cf_value =
    if struct.respond_to?(:to_cf)
      struct.to_cf
    else
      struct.to_h do |subkey, subvalue|
        subprops[subkey.to_s].to_cf(subvalue)
      end
    end

  [key, cf_value]
end
