def to_cf(struct)
  cf_value =
    if struct.respond_to?(:to_cf)
      struct.to_cf
    else
      struct.map do |subkey, subvalue|
        subprops[subkey.to_s].to_cf(subvalue)
      end.to_h
    end

  [key, cf_value]
end
