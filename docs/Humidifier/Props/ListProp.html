def to_cf(list)
  cf_value =
    if list.respond_to?(:to_cf)
      list.to_cf
    else
      list.map { |value| subprop.to_cf(value).last }
    end

  [key, cf_value]
end

def valid?(list)
  return true if whitelisted_value?(list)
  list.is_a?(Enumerable) && list.all? { |value| subprop.valid?(value) }
end
