def to_cf(map)
  cf_value =
    if map.respond_to?(:to_cf)
      map.to_cf
    else
      map.to_h do |subkey, subvalue|
        [subkey, subprop.to_cf(subvalue).last]
      end
    end

  [key, cf_value]
end

def valid?(map)

  map.is_a?(Hash) && map.values.all? { |value| subprop.valid?(value) }
