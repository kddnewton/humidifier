def initialize
  @shim =
    if Humidifier.config.sdk_version_1?
      AwsAdapters::SDKV1.new
    elsif Humidifier.config.sdk_version_2?
      AwsAdapters::SDKV2.new
    elsif Humidifier.config.sdk_version_3?
      AwsAdapters::SDKV3.new
    else
      guess_sdk
    end
end
