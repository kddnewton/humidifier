def load
  parsed = parse_spec
  types = PropertyTypes.new(parsed['PropertyTypes'])

  parsed['ResourceTypes'].each do |key, spec|
    match = key.match(/\A(\w+)::(\w+)::(\w+)\z/)
    register(match[1], match[2], match[3], spec, types.search(key))
  end

  Humidifier.registry.freeze
end
