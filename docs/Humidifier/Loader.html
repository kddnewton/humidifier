def load
  parsed   = JSON.parse(File.read(SPECPATH))
  structs  = StructureContainer.new(parsed['PropertyTypes'])

  parsed['ResourceTypes'].each do |key, spec|
    match = key.match(/\AAWS::(\w+)::(\w+)\z/)
    register(match[1], match[2], spec, structs.search(key))
  end
end
